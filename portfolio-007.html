⚙ Google Apps Script Setup

Set this up once and live prices will work on desktop + mobile — same as your =ImportHTML formula in Google Sheets.
1 Go to script.google.com → New Project
2 Paste this code:
function doGet(e) { var html = UrlFetchApp.fetch( 'https://www.merolagani.com/LatestMarket.aspx', {muteHttpExceptions: true, headers: {'User-Agent': 'Mozilla/5.0'}} ).getContentText(); var prices = {}; var rows = html.match(/<tr[\s\S]*?<\/tr>/gi) || []; rows.forEach(function(row) { var cells = row.match(/<td[^>]*>([\s\S]*?)<\/td>/gi) || []; if (cells.length < 4) return; var g = function(c) { return c.replace(/<[^>]+>/g,'').replace(/&nbsp;/g,' ').trim(); }; var sym = g(cells[0]).toUpperCase().replace(/\s/g,''); var ltp = parseFloat(g(cells[1]).replace(/,/g,'')); var chg = g(cells[2]); var prev = parseFloat(g(cells[6]||'').replace(/,/g,'')); if (sym && ltp > 0 && sym.length >= 2 && sym.length <= 12 && !/^\d+$/.test(sym)) prices[sym] = {ltp: ltp, chg: chg, chgPt: isNaN(prev)?'—':(ltp-prev).toFixed(2)}; }); return ContentService.createTextOutput(JSON.stringify(prices)) .setMimeType(ContentService.MimeType.JSON); }
3 Deploy → New Deployment → Type: Web App → Execute as: Me → Access: Anyone → Deploy → copy URL
